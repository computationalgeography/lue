add_library(lue_vulkan STATIC
    src/application_info.cpp
    src/debug_report_callback.cpp
    src/error.cpp
    src/instance.cpp

    src/physical_device.cpp
)

        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
)

target_link_libraries(lue_vulkan
    PUBLIC
        Vulkan::Vulkan
)
